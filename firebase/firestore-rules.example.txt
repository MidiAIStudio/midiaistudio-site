rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    function signedIn() {
      return request.auth != null;
    }

    function isOwner(uid) {
      return signedIn() && request.auth.uid == uid;
    }

    function isAdmin() {
      return signedIn()
        && exists(/databases/$(database)/documents/users/$(request.auth.uid))
        && get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == "admin";
    }

    match /users/{userId} {
      allow read, write: if isOwner(userId) || isAdmin();
    }

    match /licenses/{userId} {
      allow read: if isOwner(userId) || isAdmin();
      allow write: if isAdmin();
    }

    match /orders/{orderId} {
      allow read: if isAdmin() || (signedIn() && resource.data.uid == request.auth.uid);
      allow write: if isAdmin();
    }

    match /announcements/{postId} {
      allow read: if resource.data.visible == true || isAdmin();
      allow write: if isAdmin();
    }

    match /patchNotes/{postId} {
      allow read: if resource.data.visible == true || isAdmin();
      allow write: if isAdmin();
    }

    match /faq/{faqId} {
      allow read: if resource.data.visible == true || isAdmin();
      allow write: if isAdmin();
    }

    match /downloads/{downloadId} {
      allow read: if true;
      allow write: if isAdmin();
    }

    match /supportTickets/{ticketId} {
      allow create: if signedIn()
        && request.resource.data.uid == request.auth.uid;

      allow read: if isAdmin()
        || (signedIn() && resource.data.uid == request.auth.uid);

      allow update: if isAdmin()
        || (signedIn() && resource.data.uid == request.auth.uid);

      allow delete: if isAdmin()
        || (signedIn() && resource.data.uid == request.auth.uid);

      match /replies/{replyId} {
        allow read: if isAdmin()
          || (signedIn()
          && get(/databases/$(database)/documents/supportTickets/$(ticketId)).data.uid == request.auth.uid);

        allow create: if signedIn()
          && (
            isAdmin()
            || get(/databases/$(database)/documents/supportTickets/$(ticketId)).data.uid == request.auth.uid
          );

        allow update, delete: if isAdmin();
      }
    }

    match /boardPosts/{postId} {
      allow read: if resource.data.visible == true || isAdmin();
      allow create: if signedIn()
        && request.resource.data.uid == request.auth.uid;
      allow update: if isAdmin()
        || (signedIn() && resource.data.uid == request.auth.uid);
      allow delete: if isAdmin()
        || (signedIn() && resource.data.uid == request.auth.uid);

      match /comments/{commentId} {
        allow read: if get(/databases/$(database)/documents/boardPosts/$(postId)).data.visible == true || isAdmin();
        allow create: if signedIn()
          && request.resource.data.uid == request.auth.uid;
        allow update: if isAdmin()
          || (signedIn() && resource.data.uid == request.auth.uid);
        allow delete: if isAdmin()
          || (signedIn() && resource.data.uid == request.auth.uid);
      }

      match /likes/{userId} {
        allow read: if signedIn() && request.auth.uid == userId;
        allow create, delete: if signedIn() && request.auth.uid == userId;
        allow update: if false;
      }
    }

  }
}
